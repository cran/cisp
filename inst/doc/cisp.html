A correlation indicator based on spatial patterns

Install and load R packages

if (!requireNamespace("devtools")) {
    install.packages("devtools")
}
devtools::install_github("stscl/cisp",
                         build_vignettes = TRUE,
                         dep = TRUE)
library(cisp)
